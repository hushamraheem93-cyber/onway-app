// 2. كود تصميم كرت القسم المتحرك (The Animated Gradient Card)
Widget buildOnWayCard(int index) {
  return InkWell(
    onTap: () { /* الانتقال للقسم */ },
    child: AnimatedContainer(
      duration: Duration(milliseconds: 300),
      curve: Curves.easeInOut,
      margin: EdgeInsets.all(8),
      decoration: BoxDecoration(
        borderRadius: BorderRadius.circular(25), // حواف ناعمة جداً
        gradient: LinearGradient(
          begin: Alignment.topCenter,
          end: Alignment.bottomCenter,
          colors: [
            onWayCategories[index]['color'].withOpacity(0.4), // تدرج من لون القسم
            Colors.white,                                    // يتلاشى للأبيض
          ],
        ),
        boxShadow: [
          BoxShadow(
            color: Colors.black.withOpacity(0.04), // ظل ناعم جداً
            blurRadius: 10,
            offset: Offset(0, 5),
          ),
        ],
      ),
      child: Column(
        mainAxisAlignment: MainAxisAlignment.center,
        children: [
          // صور 3D Render بخلفية شفافة (سر المظهر الاحترافي)
          Expanded(
            child: Padding(
              padding: const EdgeInsets.all(12.0),
              child: Image.asset(onWayCategories[index]['icon'], fit: BoxFit.contain),
            ),
          ),
          Padding(
            padding: const EdgeInsets.only(bottom: 12.0),
            child: Text(
              onWayCategories[index]['title'],
              textAlign: TextAlign.center,
              style: TextStyle(
                color: Colors.grey[800],
                fontWeight: FontWeight.bold,
                fontSize: 14,
                height: 1.2,
              ),
            ),
          ),
        ],
      ),
    ),
  );
}
