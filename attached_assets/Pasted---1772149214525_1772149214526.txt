// 1. [المنطق البرمجي للربط وقاعدة البيانات]
// يتم التحقق من جدول "user_redemptions" في القاعدة للتأكد من عدم التكرار

const checkPromoEligibility = async (userId, promoCode, currentTotal) => {
    // أ- جلب بيانات الكود من جدول (promoCodes)
    const promo = await db.collection('promoCodes').doc(promoCode).get();
    
    if (!promo.exists()) return { status: 'error', message: 'الكود غير موجود' };
    const promoData = promo.data();

    // ب- التحقق من جدول (usageHistory) - هل المستخدم استخدمه سابقاً؟
    const alreadyUsed = await db.collection('usageHistory')
        .where('userId', '==', userId)
        .where('promoCode', '==', promoCode)
        .get();

    if (!alreadyUsed.empty) {
        return { status: 'error', message: 'لقد استخدمت هذا الكود مسبقاً!' };
    }

    // ج- حساب الخصم وتحديث السعر
    let discount = promoData.type === 'percentage' 
        ? currentTotal * (promoData.value / 100) 
        : promoData.value;

    return {
        status: 'success',
        discountAmount: discount,
        newTotal: currentTotal - discount
    };
};

// 3. [واجهة تطبيق الزبون - Checkout Screen]
const CheckoutPage = ({ userId, cartTotal }) => {
    const handleApply = async (code) => {
        const result = await checkPromoEligibility(userId, code, cartTotal);
        
        if (result.status === 'success') {
            // تحديث السعر في الواجهة
            updateTotal(result.newTotal);
            
            // ملاحظة للايجنت: عند إتمام الطلب بنجاح، يجب تسجيل العملية في جدول (usageHistory)
            // db.collection('usageHistory').add({ userId, promoCode: code, date: new Date() });
            
            alert(`تم الخصم! السعر الجديد: ${result.newTotal} دينار`);
        } else {
            alert(result.message);
        }
    };

    return (
        <View style={styles.promoContainer}>
            <TextInput placeholder="أدخل كود التخفيض" />
            <TouchableOpacity onPress={() => handleApply(inputCode)}>
