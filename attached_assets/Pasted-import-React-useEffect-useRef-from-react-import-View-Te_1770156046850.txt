import React, { useEffect, useRef } from 'react';
import { View, Text, TextInput, StyleSheet, Image, Animated, Easing } from 'react-native';

const AnimatedLogin = () => {
  // قيم الأنيميشن
  const fadeAnim = useRef(new Animated.Value(0)).current; 
  const scaleAnim = useRef(new Animated.Value(0.5)).current;
  const floatAnim = useRef(new Animated.Value(0)).current;

  useEffect(() => {
    // أنيميشن ظهور العلم
    Animated.parallel([
      Animated.timing(fadeAnim, {
        toValue: 1,
        duration: 1000,
        useNativeDriver: true,
      }),
      Animated.spring(scaleAnim, {
        toValue: 1,
        friction: 4,
        useNativeDriver: true,
      })
    ]).start();

    // أنيميشن طفو الشعار (حركة مستمرة)
    Animated.loop(
      Animated.sequence([
        Animated.timing(floatAnim, {
          toValue: -10,
          duration: 2000,
          easing: Easing.inOut(Easing.sin),
          useNativeDriver: true,
        }),
        Animated.timing(floatAnim, {
          toValue: 0,
          duration: 2000,
          easing: Easing.inOut(Easing.sin),
          useNativeDriver: true,
        })
      ])
    ).start();
  }, []);

  return (
    <View style={styles.container}>
      {/* الشعار مع حركة الطفو */}
      <Animated.View style={[styles.logoBox, { transform: [{ translateY: floatAnim }] }]}>
        <Text style={styles.logoTitle}>Onway</Text>
        <Text style={styles.logoSub}>أون وي</Text>
      </Animated.View>

      <View style={styles.formCard}>
        <Text style={styles.label}>رقم الهاتف</Text>
        
        <View style={styles.inputRow}>
          <TextInput 
            placeholder="7xx xxx xxxx"
            keyboardType="phone-pad"
            style={styles.input}
          />
          
          <View style={styles.divider} />

          {/* العلم العراقي مع أنيميشن الظهور والسكيل */}
          <Animated.View style={[
            styles.countryBox, 
            { opacity: fadeAnim, transform: [{ scale: scaleAnim }] }
          ]}>
            <Text style={styles.code}>+964</Text>
            <Image 
              source={{ uri: 'https://flagcdn.com/w80/iq.png' }} 
              style={styles.flag} 
            />
          </Animated.View>
        </View>
      </View>
    </View>
  );
};

const styles = StyleSheet.create({
  container: { flex: 1, backgroundColor: '#FF6B00', justifyContent: 'center', padding: 25 },
  logoBox: { alignItems: 'center', marginBottom: 60 },
  logoTitle: { fontSize: 50, fontWeight: 'bold', color: '#FFF', letterSpacing: 2 },
  logoSub: { fontSize: 18, color: '#FFF', opacity: 0.9 },
  formCard: { backgroundColor: 'rgba(255,255,255,0.1)', padding: 20, borderRadius: 25 },
  label: { color: '#FFF', textAlign: 'right', marginBottom: 10 },
  inputRow: { 
    flexDirection: 'row', 
    backgroundColor: '#FFF', 
    borderRadius: 15, 
    height: 65, 
    alignItems: 'center', 
    paddingHorizontal: 15 
  },
  input: { flex: 1, fontSize: 18, textAlign: 'left' },
  divider: { width: 1, height: '50%', backgroundColor: '#DDD', marginHorizontal: 10 },
  countryBox: { flexDirection: 'row', alignItems: 'center' },
  code: { fontSize: 16, fontWeight: 'bold', marginRight: 5 },
  flag: { width: 30, height: 20, borderRadius: 3 }
});

export default AnimatedLogin;
