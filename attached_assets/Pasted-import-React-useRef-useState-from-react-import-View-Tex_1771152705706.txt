import React, { useRef, useState } from 'react';
import {
  View,
  Text,
  StyleSheet,
  TouchableOpacity,
  Dimensions,
} from 'react-native';
import { SafeAreaView } from 'react-native-safe-area-context';
import { LinearGradient } from 'expo-linear-gradient';
import PagerView from 'react-native-pager-view';
import { useNavigation } from '@react-navigation/native';
import type { NativeStackNavigationProp } from '@react-navigation/native-stack';

const { width, height } = Dimensions.get('window');

type RootStackParamList = {
  PhoneLogin: { userType: 'customer' | 'driver' };
};

type NavigationProp = NativeStackNavigationProp<RootStackParamList>;

const slides = [
  {
    id: 1,
    image: '🛵',
    title: 'توصيل سريع وموثوق',
    description: 'احصل على طلبك في أي مكان وأي وقت داخل قضاء الضلوعية',
  },
  {
    id: 2,
    image: '🛒',
    title: 'تسوق بسهولة',
    description: 'اختر من مئات المنتجات المتوفرة بأفضل الأسعار',
  },
  {
    id: 3,
    image: '📦',
    title: 'تتبع طلبك',
    description: 'تابع طلبك لحظة بلحظة حتى يصل إليك',
  },
];

export default function OnboardingScreen() {
  const navigation = useNavigation<NavigationProp>();
  const pagerRef = useRef<PagerView>(null);
  const [currentPage, setCurrentPage] = useState(0);

  const handleNext = () => {
    if (currentPage < slides.length - 1) {
      pagerRef.current?.setPage(currentPage + 1);
    } else {
      navigation.navigate('PhoneLogin', { userType: 'customer' });
    }
  };

  return (
    <SafeAreaView style={styles.container}>
      <LinearGradient
        colors={['#FF6B35', '#FF8C61']}
        style={styles.gradient}
      >
        {/* Logo */}
        <View style={styles.logoContainer}>
          <View style={styles.logoBox}>
            <Text style={styles.logoIcon}>🚚</Text>
          </View>
          <View style={styles.logoTextContainer}>
            <Text style={styles.logoOn}>On</Text>
            <Text style={styles.logoWay}>Way</Text>
          </View>
        </View>

        {/* Illustration */}
        <View style={styles.illustrationContainer}>
          <View style={styles.circle}>
            <Text style={styles.illustration}>
              {slides[currentPage].image}
            </Text>
          </View>
        </View>

        {/* Content Card */}
        <View style={styles.contentCard}>
          {/* Page Indicator */}
          <View style={styles.indicatorContainer}>
            {slides.map((_, index) => (
              <View
                key={index}
                style={[
                  styles.indicator,
                  currentPage === index && styles.activeIndicator,
                ]}
              />
            ))}
          </View>

          {/* Title & Description */}
          <Text style={styles.title}>{slides[currentPage].title}</Text>
          <Text style={styles.description}>
            {slides[currentPage].description}
          </Text>

          {/* Dots (visual only) */}
          <View style={styles.dotsContainer}>
            {slides.map((_, index) => (
              <View
                key={index}
                style={[
                  styles.dot,
                  currentPage === index && styles.activeDot,
                ]}
              />
            ))}
          </View>

          {/* Next Button */}
          <TouchableOpacity
            style={styles.button}
            onPress={handleNext}
            activeOpacity={0.9}
          >
            <LinearGradient
              colors={['#FF6B35', '#FF8C61']}
              style={styles.buttonGradient}
              start={{ x: 0, y: 0 }}
              end={{ x: 1, y: 0 }}
            >
              <Text style={styles.buttonText}>
                {currentPage === slides.length - 1 ? 'ابدأ الآن' : 'التالي'}
              </Text>
              <Text style={styles.buttonIcon}>←</Text>
            </LinearGradient>
          </TouchableOpacity>
        </View>

        {/* Hidden PagerView for state management */}
        <PagerView
          ref={pagerRef}
          style={styles.hiddenPager}
          initialPage={0}
          onPageSelected={(e) => setCurrentPage(e.nativeEvent.position)}
        >
          {slides.map((slide) => (
            <View key={slide.id} />
          ))}
        </PagerView>
      </LinearGradient>
    </SafeAreaView>
  );
}
