import { db } from './firebase';
import { 
  collection, 
  doc, 
  getDocs, 
  getDoc,
  updateDoc, 
  query, 
  where,
  orderBy,
  addDoc,
  Timestamp,
} from 'firebase/firestore';
import type { Order, DeliveryBatch, DriverStats } from '../types/driver';

const ORDERS_COLLECTION = 'orders';
const BATCHES_COLLECTION = 'delivery_batches';
const DRIVERS_COLLECTION = 'drivers';

// ========================================
// الحصول على إحصائيات السائق
// ========================================
export const getDriverStats = async (driverId: string): Promise<DriverStats> => {
  try {
    const driverRef = doc(db, DRIVERS_COLLECTION, driverId);
    const driverDoc = await getDoc(driverRef);
    
    if (!driverDoc.exists()) {
      throw new Error('السائق غير موجود');
    }
    
    const data = driverDoc.data();
    
    return {
      activeOrdersCount: data.active_orders_count || 0,
      maxOrdersPerBatch: data.max_orders_per_batch || 5,
      totalCompletedOrders: data.total_completed_orders || 0,
      totalEarnings: data.total_earnings || 0,
      currentBatchId: data.current_batch_id,
    };
  } catch (error) {
    console.error('خطأ في جلب إحصائيات السائق:', error);
    throw error;
  }
};

// ========================================
// الحصول على الطلبات النشطة للسائق
// ========================================
export const getActiveOrders = async (driverId: string): Promise<Order[]> => {
  try {
    const ordersRef = collection(db, ORDERS_COLLECTION);
    const q = query(
      ordersRef,
      where('driver_id', '==', driverId),
      where('status', 'in', ['picked_up', 'in_delivery']),
      orderBy('delivery_sequence', 'asc')
    );
    
    const snapshot = await getDocs(q);
    const orders: Order[] = [];
    
    snapshot.forEach((doc) => {
      const data = doc.data();
      orders.push({
        id: doc.id,
        orderNumber: data.order_number,
        restaurantId: data.restaurant_id,
        restaurantName: data.restaurant_name,
        restaurantAddress: data.restaurant_address,
        restaurantPhone: data.restaurant_phone,
        customerId: data.customer_id,
        customerName: data.customer_name,
        customerAddress: data.customer_address,
        customerPhone: data.customer_phone,
        customerLat: data.customer_lat,
        customerLng: data.customer_lng,
        items: data.items,
        totalPrice: data.total_price,
        deliveryFee: data.delivery_fee,
        status: data.status,
        driverId: data.driver_id,
        batchId: data.batch_id,
        deliverySequence: data.delivery_sequence,
        distance: data.distance,
        estimatedTime: data.estimated_time,
        notes: data.notes,
        pickedUpAt: data.picked_up_at?.toDate(),
        deliveredAt: data.delivered_at?.toDate(),
        createdAt: data.created_at.toDate(),
      });
    });
    
    return orders;
  } catch (error) {
    console.error('خطأ في جلب الطلبات النشطة:', error);
    throw error;
  }
};

// ========================================
// الحصول على الطلبات المتاحة
// ========================================
export const getAvailableOrders = async (
  driverLat: number,
  driverLng: number,
  maxDistance: number = 10
): Promise<Order[]> => {
  try {
    const ordersRef = collection(db, ORDERS_COLLECTION);
    const q = query(
      ordersRef,
      where('status', '==', 'ready'),
      where('driver_id', '==', null),
      orderBy('created_at', 'desc')
    );
    
    const snapshot = await getDocs(q);
    const orders: Order[] = [];
    
    snapshot.forEach((doc) => {
      const data = doc.data();
      
      // حساب المسافة
      const distance = calculateDistance(
        driverLat,
        driverLng,
        data.customer_lat,
        data.customer_lng
      );
      
      // فقط الطلبات القريبة
      if (distance <= maxDistance) {
        orders.push({
          id: doc.id,
          orderNumber: data.order_number,
          restaurantId: data.restaurant_id,
          restaurantName: data.restaurant_name,
          restaurantAddress: data.restaurant_address,
          restaurantPhone: data.restaurant_phone,
          customerId: data.customer_id,
          customerName: data.customer_name,
          customerAddress: data.customer_address,
          customerPhone: data.customer_phone,
          customerLat: data.customer_lat,
          customerLng: data.customer_lng,
          items: data.items,
          totalPrice: data.total_price,
          deliveryFee: data.delivery_fee,
          status: data.status,
          deliverySequence: 0,
          distance: parseFloat(distance.toFixed(1)),
          estimatedTime: calculateEstimatedTime(distance),
          createdAt: data.created_at.toDate(),
        });
      }
    });
    
    // ترتيب حسب الأقرب
    orders.sort((a, b) => a.distance - b.distance);
    
    return orders;
  } catch (error) {
    console.error('خطأ في جلب الطلبات المتاحة:', error);
    throw error;
  }
};

// ========================================
// استلام طلب جديد
// ========================================
export const acceptOrder = async (
  orderId: string,
  driverId: string
): Promise<void> => {
  try {
    // التحقق من عدد الطلبات النشطة
    const stats = await getDriverStats(driverId);
    
    if (stats.activeOrdersCount >= stats.maxOrdersPerBatch) {
      throw new Error(
        `لا يمكنك استلام أكثر من ${stats.maxOrdersPerBatch} طلبات في نفس الوقت`
      );
    }
    
    // الحصول على رقم الترتيب
    const activeOrders = await getActiveOrders(driverId);
    const deliverySequence = activeOrders.length + 1;
    
    // تحديث الطلب
    const orderRef = doc(db, ORDERS_COLLECTION, orderId);
    await updateDoc(orderRef, {
      driver_id: driverId,
      status: 'picked_up',
      delivery_sequence: deliverySequence,
      picked_up_at: Timestamp.now(),
      updated_at: Timestamp.now(),
    });
    
    // تحديث إحصائيات السائق
    const driverRef = doc(db, DRIVERS_COLLECTION, driverId);
    await updateDoc(driverRef, {
      active_orders_count: stats.activeOrdersCount + 1,
      updated_at: Timestamp.now(),
    });
    
    // تسجيل الإجراء
    await addDoc(collection(db, 'delivery_logs'), {
      order_id: orderId,
      driver_id: driverId,
      action: 'picked_up',
      created_at: Timestamp.now(),
    });
    
    console.log('✅ تم استلام الطلب بنجاح');
  } catch (error) {
    console.error('❌ خطأ في استلام الطلب:', error);
    throw error;
  }
};

// ========================================
// إكمال توصيل طلب
// ========================================
export const completeOrder = async (
  orderId: string,
  driverId: string
): Promise<void> => {
  try {
    // تحديث الطلب
    const orderRef = doc(db, ORDERS_COLLECTION, orderId);
    await updateDoc(orderRef, {
      status: 'delivered',
      delivered_at: Timestamp.now(),
      updated_at: Timestamp.now(),
    });
    
    // الحصول على تفاصيل الطلب لحساب الأرباح
    const orderDoc = await getDoc(orderRef);
    const orderData = orderDoc.data();
    const deliveryFee = orderData?.delivery_fee || 0;
    
    // تحديث إحصائيات السائق
    const stats = await getDriverStats(driverId);
    const driverRef = doc(db, DRIVERS_COLLECTION, driverId);
    await updateDoc(driverRef, {
      active_orders_count: Math.max(0, stats.activeOrdersCount - 1),
      total_completed_orders: stats.totalCompletedOrders + 1,
      total_earnings: stats.totalEarnings + deliveryFee,
      updated_at: Timestamp.now(),
    });
    
    // تسجيل الإجراء
    await addDoc(collection(db, 'delivery_logs'), {
      order_id: orderId,
      driver_id: driverId,
      action: 'delivered',
      created_at: Timestamp.now(),
    });
    
    console.log('✅ تم توصيل الطلب بنجاح');
  } catch (error) {
    console.error('❌ خطأ في إكمال الطلب:', error);
    throw error;
  }
};

// ========================================
// حساب المسافة بين نقطتين (بالكيلومتر)
// ========================================
const calculateDistance = (
  lat1: number,
  lng1: number,
  lat2: number,
  lng2: number
): number => {
  const R = 6371; // نصف قطر الأرض بالكيلومتر
  const dLat = toRad(lat2 - lat1);
  const dLng = toRad(lng2 - lng1);
  
  const a =
    Math.sin(dLat / 2) * Math.sin(dLat / 2) +
    Math.cos(toRad(lat1)) *
      Math.cos(toRad(lat2)) *
      Math.sin(dLng / 2) *
      Math.sin(dLng / 2);
  
  const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
  const distance = R * c;
  
  return distance;
};

const toRad = (value: number): number => {
  return (value * Math.PI) / 180;
};

// ========================================
// حساب الوقت المتوقع (بالدقائق)
// ========================================
const calculateEstimatedTime = (distance: number): string => {
  // سرعة متوسطة 30 كم/ساعة
  const timeInHours = distance / 30;
  const timeInMinutes = Math.ceil(timeInHours * 60);
  
  if (timeInMinutes < 60) {
    return `${timeInMinutes} دقيقة`;
  } else {
    const hours = Math.floor(timeInMinutes / 60);
    const minutes = timeInMinutes % 60;
    return `${hours} ساعة و ${minutes} دقيقة`;
  }
};

// ========================================
// ترتيب الطلبات حسب المسار الأمثل
// ========================================
export const optimizeDeliveryRoute = (
  orders: Order[],
  startLat: number,
  startLng: number
): Order[] => {
  if (orders.length === 0) return [];
  
  const optimized: Order[] = [];
  const remaining = [...orders];
  let currentLat = startLat;
  let currentLng = startLng;
  
  // خوارزمية Nearest Neighbor (الأقرب أولاً)
  while (remaining.length > 0) {
    let nearestIndex = 0;
    let shortestDistance = calculateDistance(
      currentLat,
      currentLng,
      remaining[0].customerLat,
      remaining[0].customerLng
    );
    
    for (let i = 1; i < remaining.length; i++) {
      const distance = calculateDistance(
        currentLat,
        currentLng,
        remaining[i].customerLat,
        remaining[i].customerLng
      );
      
      if (distance < shortestDistance) {
        shortestDistance = distance;
        nearestIndex = i;
      }
    }
    
    const nearest = remaining.splice(nearestIndex, 1)[0];
    optimized.push({
      ...nearest,
      deliverySequence: optimized.length + 1,
    });
    
    currentLat = nearest.customerLat;
    currentLng = nearest.customerLng;
  }
  
  return optimized;
};
